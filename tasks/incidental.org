  Quick scrip to test if font is monospaced using /python/ with =sys= and =fontforge=

  import sys
  import fontforge

  f = fontforge.open(sys.argv[1])
  i = f["i"]
  m = f["m"]

  if i.width == m.width:
      print("Monospace!")
  #+END_SRC
