#+name:fzf_notifications
#+BEGIN_SRC sh :tangle yes :noweb yes
#!/usr/bin/env bash
# FZF-based notification viewer with clipboard copy
# Format: $TITLE - $SUMMARY

# Get notification history as JSON
history_json=$(dunstctl history)

# Parse JSON and format for fzf
# Each line: "TITLE - SUMMARY"
echo "$history_json" | jq -r '
  .data[] | 
  .[] | 
  select(.summary != null) |
  "\(.summary.data) - \(.body.data // "")"
' | awk 'NF > 0' | fzf --prompt="Select notification: " \
  --preview='echo "Selected: {}"' \
  --bind='enter:execute(echo "{}" | wl-copy)+abort' \
  --header='Press Enter to copy to clipboard'
#+END_SRC

/Sending a notification on systemd service failure/
-----

Achieving by:
- Creating a service =notify-on-failure@= that send notification via =notify-send=
- Run the service =notify-on-failure@%n= on main service failure
  + =%n= parse the main service name
  + =%i= in notification service parse whole main service name as variable
- Send notification with failed service name

#+name: notifi-on-failure@.service
#+begin_src systemd
[Unit]
Description=Notify on service failure - receives failed unit name via instance

[Service]
Type=oneshot
ExecStart=/usr/bin/dunstify -u critical "Systemd" "Service Failed: Unit '%i' has failed — check logs for details"
#+end_src
