  DEVICE=${1}
  STATE=${2}

  Resolved with systemd =drop-in= e.g.
  + Original service:
    #+name: original service
    #+begin_src systemd
    [Unit]
    Description=ntfy client
    After=network.target

    [Service]
    ExecStart=/usr/bin/ntfy subscribe --config "%h/.config/ntfy/client.yml" --from-config
    Restart=on-failure

    [Install]
    WantedBy=default.target
    #+end_src

#+name:Original Service
#+begin_quote markdown
[!NOTE]
The service is manage by upstream =ntfy.sh= via =pacman= and update frequently
#+end_quote

  + Modified version:
    #+name: modified service
    #+begin_src systemd
    
    [Unit]
    Description=ntfy client
    After=internet.target
    BindsTo=internet.target

    [Service]
    ExecStart=/usr/bin/bash -c '/usr/bin/ntfy subscribe --from-config --since $(/usr/bin/cat ~/.cache/ntfy/downtime)'
    ExecStop=/usr/bin/bash -c '/usr/bin/date +%%s > ~/.cache/ntfy/downtime'
    Restart=on-failure

    [Install]
    WantedBy=qtile-session.target internet.target
    #+end_src
    
#+name:modified service update warn
#+begin_quote markdown
[!WARNING]
This service will not get any updates from upstream since it is managed locally
#+end_quote

  + Drop-ins version:
    #+name: drop-in version
    #+begin_src systemd
    [Unit]
    After=internet.target
    BindsTo=internet.target

    [Service]
    ExecStartPre=/usr/bin/bash -c '/usr/bin/ntfy subscribe --from-config --poll=true --since $(/usr/bin/cat ~/.cache/ntfy/downtime)'
    ExecStop=/usr/bin/bash -c '/usr/bin/date +%%s > ~/.cache/ntfy/downtime'
    Restart=on-failure

    [Install]
    WantedBy=qtile-session.target internet.target
    #+end_src
    This services get updates from upstream and behave exactly how it should defined parameter
